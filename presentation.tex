% FRAME 6 %
\begin{frame}
  \frametitle{Type Systems}
  \begin{itemize}
  \item Simply Typed $\lambda$-calculus by Alonzo Church
  \item Static Typing - simple solution to catch type errors at compile time.
  \item Hindley-Milner - introduced type inference and polymorphism
  \item Advanced Type Systems - type classes, GADTs, higher-kinded types, OOP, etc.
  \end{itemize}
\end{frame}

% FRAME 7 %
\begin{frame}
  \frametitle{Dependent Types}
  \begin{itemize}
  \item Types depend on terms (values).
  \item Types are first-class citizens.
  \item Classic example: Array Length
  \item Static typing with dynamic properties (not true).
  \end{itemize}
\end{frame}
